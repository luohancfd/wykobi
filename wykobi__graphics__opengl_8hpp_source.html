213  inline void draw_circle(const T& x, const T& y, const T& radius) const
214  {
215  wykobi::polygon<T,2> _polygon = make_polygon(make_circle(x,y,radius));
216  glColor3fv (color);
217  switch (drawing_mode)
218  {
219  case eSolid : glBegin(GL_TRIANGLE_FAN); break;
220  case eOutLine : glBegin(GL_LINE_LOOP); break;
221  case eNoDraw : return;
222  }
223  if (drawing_mode == eSolid)
224  {
225  glVertex2d(double(_polygon[0].x),double(_polygon[0].y));
226  }
227  for (std::size_t i = (drawing_mode == eSolid)?1:0; i < _polygon.size(); ++i)
228  {
229  glVertex2d(double(_polygon[i].x),double(_polygon[i].y));
230  }
231  glEnd();

335  inline void draw(const polygon<T,2>& polygon, const bool convex = false)
336  {
337  if (polygon.size() < 3) return;
338  if (!convex)
339  {
340  std::size_t j = polygon.size() - 1;
341  for (std::size_t i = 0; i < polygon.size(); ++i)
342  {
343  draw_segment(polygon[i],polygon[j]);
344  j = i;
345  }
346  }
347  else
348  {
349  switch (drawing_mode)
350  {
351  case eSolid : glBegin(GL_POLYGON); break;
352  case eOutLine : glBegin(GL_LINE_LOOP); break;
353  case eNoDraw : return;
354  }
355  for (std::size_t i = 0; i < polygon.size(); ++i)
356  {
357  glVertex3d(polygon[i].x1,polygon[i].y1,polygon[i].z1);
